function initItems() {
getItems().forEach(function(item) {

02 — Animation de hauteur

Le problème de height:auto — impossible à animer directement en CSS ¶

En CSS, on ne peut pas écrire transition: height 0.3s de 0 vers auto. Le navigateur ne sait pas calculer le point d'arrivée. La solution : mesurer la hauteur réelle en JS, puis animer vers cette valeur fixe.

openItem() — les 4 étapes pour animer height:0 height:auto
①
height = 'auto' le navigateur calcule la vraie hauteur
②
scrollHeight lu ex: 148px (valeur fixe mémorisée)
③
height = '0' retour à 0, instantané (pas de transition)
④
requestAnimationFrame height = '148px' CSS transition s'active : 0 148px en 0.3s
Le passage par height='0' avant rAF est obligatoire. Sans lui, le navigateur ne voit pas de changement d'état et n'anime pas la transition.
javascript — openItem() — la technique en 4 étapes
function openItem(item) {
const answer = item.querySelector('.faq-answer');
item.classList.add('active');
// Étape 1 : passe en height:auto pour MESURER answer.style.height = 'auto';
const targetHeight = answer.scrollHeight + 'px'; // Ex: "248px" // Étape 2 : remet à 0 pour partir du bon point de départ answer.style.height = '0';
// Étape 3 : FORCE le navigateur à recalculer le DOM (reflow) // Sans ça, le navigateur optimise et saute l'étape 2  pas d'animation answer.getBoundingClientRect();
// Étape 4 : installe la transition et lance l'animation answer.style.transition = 'height 0.32s cubic-bezier(0.4, 0, 0.2, 1)';
answer.style.height     = targetHeight; //  animé de 0 vers 248px // Après l'animation : repasse en auto pour s'adapter au contenu dynamique answer.addEventListener('transitionend', function onEnd() {
answer.removeEventListener('transitionend', onEnd); // Usage unique if (item.classList.contains('active')) answer.style.height = 'auto';
});
}

javascript — closeItem() — l'inverse en 3 étapes
function closeItem(item) {
const answer = item.querySelector('.faq-answer');
// Étape 1 : fige la hauteur actuelle (si elle est en 'auto', on ne peut pas partir d'auto) answer.style.height = answer.scrollHeight + 'px';
// Étape 2 : force le reflow answer.getBoundingClientRect();
// Étape 3 : transition vers 0 (légèrement plus rapide qu'à l'ouverture) answer.style.transition = 'height 0.28s cubic-bezier(0.4, 0, 0.2, 1)';
answer.style.height     = '0';
item.classList.remove('active'); // Retire la classe immédiatement (styles CSS) // Pas besoin de transitionend ici : height:0 est la position finale souhaitée
}
getBoundingClientRect() force ce qu'on appelle un reflow — le navigateur recalcule la géométrie du DOM. Sans ça, le navigateur peut "batching" les changements de style et fusionner height:0 et height:248px en un seul saut, sans transition visible. Cet appel "force" le navigateur à valider l'état height:0 avant de lancer l'animation.

javascript — la logique de filtrage
function applyFilters() {
const term = searchInput ? searchInput.value.toLowerCase().trim() : '';
const category = document.querySelector('.category-btn.active')?.dataset.category || 'all';
let visibleCount = 0;
getItems().forEach(function(item) {
const qText = item.querySelector('.faq-question-text').textContent.toLowerCase();
const aText = item.querySelector('.faq-answer').textContent.toLowerCase();
const iCat = item.dataset.category || '';
const catOk = category === 'all' || iCat === category;
const termOk = term === '' || qText.includes(term) || aText.includes(term);
const show = catOk && termOk; // Les DEUX conditions doivent être vraies item.style.display = show ? '' : 'none';
if (!show && item.classList.contains('active')) closeItem(item); // Ferme si caché if (show) visibleCount++;
});
noResults.classList.toggle('visible', visibleCount === 0); // Message si 0 résultats
}
Logique de filtrage — pour chaque item
catOk
category === 'all' tout passe
sinon item.dataset.category === category sélectionnée
termOk
term === '' tout passe
sinon term dans la question OU la réponse
show = catOk && termOk — les deux doivent être true

Debounce — ne pas filtrer à chaque frappe

javascript
let debounceTimer;
searchInput.addEventListener('input', function() {
clearTimeout(debounceTimer);          // Annule le timer précédent debounceTimer = setTimeout(applyFilters, 200); // Lance après 200ms de silence
});
// "Le"  annulé / "Les"  annulé / "Les p"  annulé / "Les pr"  200ms  applyFilters()
// Sans debounce : applyFilters s'appellerait 6 fois pour "Les pr" au lieu d'1 fois.
clearTimeout sur un ID déjà expiré ne fait rien (pas d'erreur). Donc la première frappe fait clearTimeout(undefined) no-op, puis installe le timer. C'est un pattern parfaitement sûr.

04 — Récapitulatif

Les 7 fonctions et leurs rôles ¶

faq.js orchestre 3 fonctionnalités interdépendantes : accordéon, recherche et filtres par catégorie. Leur point de convergence est applyFilters() qui centralise la décision d'afficher ou cacher chaque item.

Les 7 fonctions — rôle et chaîne d'appel

getItems() — Array.from(faqList.querySelectorAll('.faq-item')). Getter dynamique : recalcule à chaque appel pour rester synchrone si le DOM change. Appelée par initItems() et applyFilters().

initItems() — Configure chaque item accordéon (wrapper .faq-answer-inner + listeners de clic). Appelée une seule fois à l'init.

openItem(item) — Anime l'ouverture : mesure la hauteur cible, anime de 0 à cette hauteur, repasse en auto. Appelée par le listener de clic.

closeItem(item) — Anime la fermeture : fige la hauteur courante, anime vers 0. Appelée par le listener de clic et applyFilters().

applyFilters() — Combine terme de recherche et catégorie active pour filtrer les items. Appelée par le debounce de recherche, les boutons catégorie, et l'init.

Le point technique clé — animer height:auto

javascript
// Problème : CSS ne peut pas transitionner de 0 à 'auto'
// Solution en 3 étapes :

// OUVERTURE
answer.style.height = 'auto';             // 1. Mesurer
const target = answer.scrollHeight + 'px'; // 2. Capturer
answer.style.height = '0';               // 3. Réinitialiser
answer.getBoundingClientRect();           // 4. Forcer le reflow
answer.style.height = target;            // 5. Animer 0 → target

// FERMETURE
answer.style.height = answer.scrollHeight + 'px'; // 1. Figer
answer.getBoundingClientRect();                    // 2. Forcer reflow
answer.style.height = '0';                        // 3. Animer → 0
💡
getBoundingClientRect() force un reflow (recalcul de layout) par le navigateur. Sans lui, les deux changements de height seraient regroupés en un seul batch par le moteur de rendu — l'animation ne démarrerait pas depuis 0 car le navigateur optimiserait en ne gardant que le dernier état.

07 — Ordre du code

Pourquoi chaque bloc est placé là où il est ¶

faq.js suit la progression préparer → initialiser → événements → déclencher. Le message 'aucun résultat' est créé tôt car applyFilters() le manipule. Les appels finaux déclenchent tout le système.

1
Sélection + protectiongetElementById('faq-list') + querySelectorAll('.category-btn') + if (!faqList) return
Toutes les fonctions du module utilisent faqList et categoryBtns. Le guard clause sur faqList est immédiat : sans liste FAQ, initItems() ferait null.querySelectorAll('.faq-item') → TypeError. Le check sur searchInput est optionnel (la FAQ fonctionne sans barre de recherche) — il est fait plus loin avec un simple if (searchInput), pas un return.
↓ créer le message 'aucun résultat' avant applyFilters qui le manipule
2
Création DOM préalableconst noResults = document.createElement('p') + faqList.after(noResults)
Inséré dans le DOM avant applyFilters() qui fait noResults.classList.toggle('visible'). Si noResults n'était pas encore dans le document, classList.toggle('visible', true) fonctionnerait sur un élément 'fantôme' — la classe serait ajoutée en mémoire mais l'élément serait invisible car non attaché au document. La règle : un élément doit être dans le DOM avant d'être stylé via CSS.
Si créé après le premier applyFilters() : noResults.classList.toggle('visible', visible) sur un élément hors DOM → classe ajoutée → mais display:block du CSS ne s'applique pas (l'élément n'est pas dans le document) → bug silencieux : message jamais affiché même quand il le devrait.
↓ fonction plutôt que variable pour refléter l'état actuel du DOM
3
Getter dynamiquefunction getItems() { return Array.from(faqList.querySelectorAll('.faq-item')); }
Définie comme fonction pour capturer l'état courant du DOM à chaque appel. Si c'était const items = faqList.querySelectorAll('.faq-item'), la NodeList serait figée au moment de la déclaration. Si des items FAQ étaient ajoutés dynamiquement (ex: lazy loading), ils ne seraient jamais dans items. En tant que fonction, elle recalcule à chaque appel — toujours à jour avec le DOM réel.
↓ initialiser l'interactivité de base avant les filtres qui cachent/montrent
4
Accordéonsfunction initItems() — attache les clics sur .faq-question
Avant les listeners de filtre — l'accordéon doit être cliquable avant d'être filtrable. Si on filtrait avant d'initialiser les accordéons, les items seraient cachés/montrés correctement mais les clics sur les questions ne feraient rien (listeners pas encore attachés). L'interactivité de base (ouvrir/fermer) doit être en place avant les comportements avancés (filtrer, rechercher).
Appelle getItems() (étape 3), openItem(), closeItem() (étape 5 — hoistées).
↓ détails d'animation — après la vue d'ensemble

Animation accordéonfunction openItem(item) + function closeItem(item)
Après initItems() dans l'ordre de lecture — ce sont des détails d'implémentation. Ces fonctions gèrent les transitions d'hauteur (height: 0 → scrollHeight → auto). Le getBoundingClientRect() dans openItem() force un reflow navigateur — sans lui, la transition partirait de auto (non animable) plutôt que de 0. Hoistées, donc accessibles dans initItems() écrite avant.
↓ recherche texte — fonctionnalité principale
6
Recherche + debouncesearchInput.addEventListener('input') + clearTimeout/setTimeout 200ms
Avant les filtres de catégorie — c'est la fonctionnalité la plus utilisée. Le debounce de 200ms évite d'appeler applyFilters() à chaque frappe. La variable debounceTimer est déclarée juste avant ce bloc car elle est locale à ce listener — elle doit persister entre les frappes (comme searchTimeout dans recherche.js) mais n'a pas besoin d'un scope plus large.
↓ filtres de catégorie — fonctionnalité secondaire
7
Filtres catégoriecategoryBtns.forEach(btn => btn.addEventListener('click', …))
Après la recherche — complémentaire et moins utilisé. Ces boutons filtrent par catégorie (ex: 'Livraison', 'Retours'). Leur listener appelle applyFilters() — hoistée depuis le bas. L'ordre recherche → catégorie reflète la fréquence d'utilisation : les utilisateurs cherchent par mot-clé plus souvent qu'ils ne naviguent par catégorie.
↓ tout est prêt — déclencher le système

DémarrageinitItems() + applyFilters() — appels finaux
En tout dernier — utilisent tout ce qui a été défini avant. initItems() attache tous les listeners de clic sur les questions. applyFilters() effectue le filtrage initial (terme vide, catégorie 'all') → affiche tous les items au chargement. Ces deux appels sont le 'bouton démarrer' — avant eux, tout n'est que définition.
Pattern universel : tout définir d'abord, tout déclencher en dernier. Les appels d'initialisation sont toujours les dernières lignes du fichier.

Concepts et termes techniques utilisés dans ce fichier ¶

Définitions des termes importants rencontrés dans cette documentation.

Document Object Model
Représentation en mémoire de la page HTML sous forme d'arbre. Chaque balise devient un nœud. JavaScript modifie cet arbre via getElementById(), querySelector(), innerHTML. Modifier le DOM = modifier ce que l'utilisateur voit en temps réel.
Immediately Invoked Function Expression
Fonction qui s'exécute immédiatement à sa déclaration : (function() { ... })(). Crée un scope isolé — tout ce qui est déclaré dedans reste privé et ne pollue pas window. Tous les fichiers JS du projet utilisent ce pattern.
DOMContentLoaded
Événement déclenché quand le HTML est parsé et le DOM prêt — avant images et CSS. Moment idéal pour initialiser les modules JS. Distinct de window.load qui attend tous les assets réseau.
scrollHeight
Hauteur totale d'un élément incluant la partie non visible (défilable). Toujours >= hauteur visible. Utilisé pour mesurer la hauteur réelle d'un contenu avant de l'animer : height:0 → height:scrollHeight+'px'.
debounce
Regroupe plusieurs appels rapides en un seul différé. Frappe 5 lettres en 200ms → 1 seule recherche lancée 300ms après la dernière frappe. Implémenté avec clearTimeout(timer) + timer = setTimeout(fn, 300).
dataset
Interface qui expose les attributs data-* d'un élément. el.dataset.productId lit data-product-id. Permet de stocker des données métier directement dans le HTML sans variables globales. La conversion camelCase est automatique : data-product-id → dataset.productId.
classList
Interface de manipulation des classes CSS. add(), remove(), toggle(), contains(). Plus fiable que modifier className comme une string. toggle('active', condition) ajoute ou retire selon un booléen.

FAQ accordion : animation fluide et accessibilité ¶

💡
Analogie — l'accordion physique : Un accordion musical : quand tu ouvres un soufflet, les autres se ferment pour maintenir la tension. En FAQ, c'est pareil — une seule réponse visible à la fois, les autres se replient quand une nouvelle s'ouvre.
💡
Analogie — max-height pour l'animation : Animer la hauteur d'un élément à contenu variable, c'est comme essayer d'estimer la taille d'une valise avant de l'avoir remplie. La solution : mettre max-height: 0 → max-height: 1000px. Peu importe ce qu'il y a dedans, l'animation marche. Le vrai contenu détermine la hauteur finale.
🚨 PIÈGE — Animer height au lieu de max-height
height: auto ne peut pas être animé en CSS. Passer de height: 0 à height: auto ne donne aucune animation — la hauteur change instantanément. La solution : animer max-height de 0 à une valeur supérieure à la hauteur maximale possible.
🚨 PIÈGE — Dupliquer la logique ouverture/fermeture
Avoir du code séparé pour "ouvrir" et "fermer" mène à des bugs de désynchronisation. Une seule fonction toggle(item) qui vérifie l'état courant et bascule. Si l'état est géré par une classe CSS (active), classList.toggle('active') suffit.
PRO TIP
Utilise les éléments HTML natifs <details> et <summary> pour une FAQ simple. Accessibles nativement, animables en CSS, sans JS. Réserve le JS pour les besoins avancés (animation personnalisée, état partagé).
PRO TIP
Ajoute role="region" et aria-labelledby sur chaque panneau de contenu pour lier le contenu à sa question. Les lecteurs d'écran comprennent alors la structure question/réponse.

max-height trick

Anime la hauteur variable en CSS. 0 → grande valeur fixe. La hauteur réelle détermine le final.

classList.toggle

Bascule une classe active. Une opération, deux effets : ouvrir si fermé, fermer si ouvert.

Mode exclusif

Fermer tous les autres items avant d'en ouvrir un. Option configurable selon le besoin.

aria-expanded

Indique aux lecteurs d'écran si le panneau est ouvert ou fermé. Synchronisé avec l'état JS.

// faq.js - Page FAQ PearTech
// Gère l'accordéon (ouverture/fermeture des réponses),
// la recherche par mot-clé et le filtrage par catégorie
// ============================================

(function () {
    'use strict'; // Mode strict activé

    document.addEventListener('DOMContentLoaded', function () {

        // ── Sélection des éléments du DOM ────────────────────────

        const faqList      = document.getElementById('faq-list');          // Conteneur de toutes les questions/réponses
        const searchInput  = document.getElementById('faq-search-input');  // Champ de recherche
        const categoryBtns = document.querySelectorAll('.category-btn');   // Boutons de filtre par catégorie

        if (!faqList) return; // Si la liste FAQ n'existe pas sur cette page, on arrête

        // ── Création du message "aucun résultat" ──────────────────

        const noResults = document.createElement('p'); // Crée un paragraphe
        noResults.className = 'faq-no-results';        // Classe CSS pour le style
        noResults.textContent = 'Aucune question ne correspond à votre recherche.'; // Texte affiché
        faqList.after(noResults); // Insère le message juste après la liste FAQ dans le DOM

        // ── Getter dynamique des items ────────────────────────────
        // Fonction plutôt que variable : récupère les items à chaque appel
        // pour toujours avoir la liste à jour même si le DOM change

        function getItems() {
            return Array.from(faqList.querySelectorAll('.faq-item')); // Convertit NodeList en Array
        }

        // ── Initialisation de chaque item accordéon ───────────────

        function initItems() {
            getItems().forEach(function (item) {
                const answer = item.querySelector('.faq-answer'); // Zone de réponse à afficher/cacher
                if (!answer) return; // Ignore les items mal formés

                // Vérifie si le wrapper .faq-answer-inner existe déjà dans le HTML
                let inner = answer.querySelector('.faq-answer-inner');
                if (!inner) {
                    // Compatibilité : crée le wrapper si absent du HTML
                    inner = document.createElement('div');
                    inner.className = 'faq-answer-inner'; // Classe CSS pour le padding intérieur
                    // Déplace tout le contenu existant de .faq-answer dans le nouveau wrapper
                    while (answer.firstChild) inner.appendChild(answer.firstChild);
                    answer.appendChild(inner); // Ajoute le wrapper dans .faq-answer
                }

                // Ferme toutes les réponses à l'initialisation
                answer.style.height   = '0';      // Hauteur nulle = caché
                answer.style.overflow = 'hidden'; // Masque le contenu qui dépasse
                answer.style.display  = 'block';  // Nécessaire pour que la transition height fonctionne

                const btn = item.querySelector('.faq-question'); // Bouton cliquable de la question
                if (!btn) return;

                btn.addEventListener('click', function () {
                    if (item.classList.contains('active')) {
                        closeItem(item); // L'item est ouvert : on le ferme
                    } else {
                        // L'item est fermé : on ferme tous les autres d'abord (accordéon = un seul ouvert)
                        getItems().forEach(function (other) {
                            if (other !== item && other.classList.contains('active')) {
                                closeItem(other); // Ferme les autres items ouverts
                            }
                        });
                        openItem(item); // Puis on ouvre l'item cliqué
                    }
                });
            });
        }

        // ── Animation d'ouverture ─────────────────────────────────

        function openItem(item) {
            const answer = item.querySelector('.faq-answer');
            if (!answer) return;

            item.classList.add('active'); // Ajoute la classe pour les styles CSS de l'état ouvert

            // Technique pour animer height de 0 à la vraie hauteur :
            // 1. On passe en height:auto pour mesurer la hauteur réelle du contenu
            answer.style.height = 'auto';
            const targetHeight  = answer.scrollHeight + 'px'; // Mesure la hauteur naturelle
            answer.style.height = '0'; // Remet à 0 avant d'animer

            answer.getBoundingClientRect(); // Force le reflow du navigateur (nécessaire pour que la transition démarre)
            answer.style.transition = 'height 0.32s cubic-bezier(0.4, 0, 0.2, 1)'; // Courbe d'animation Material Design
            answer.style.height     = targetHeight; // Lance l'animation vers la hauteur cible

            // Après la fin de l'animation, repasse en height:auto
            // pour que le contenu dynamique (ex: images qui se chargent) ne soit pas coupé
            answer.addEventListener('transitionend', function onEnd() {
                answer.removeEventListener('transitionend', onEnd); // Supprime le listener pour éviter les doublons
                if (item.classList.contains('active')) { // Vérifie que l'item est encore ouvert
                    answer.style.height = 'auto'; // Repasse en hauteur automatique
                }
            });
        }

        // ── Animation de fermeture ────────────────────────────────

        function closeItem(item) {
            const answer = item.querySelector('.faq-answer');
            if (!answer) return;

            // Pour animer depuis la hauteur actuelle vers 0, il faut d'abord figer la hauteur
            answer.style.height     = answer.scrollHeight + 'px'; // Fige la hauteur actuelle en pixels
            answer.style.transition = 'height 0.28s cubic-bezier(0.4, 0, 0.2, 1)'; // Légèrement plus rapide à la fermeture

            answer.getBoundingClientRect(); // Force le reflow pour que la transition parte bien de la hauteur figée
            answer.style.height = '0'; // Anime vers 0

            item.classList.remove('active'); // Retire la classe pour les styles CSS de l'état fermé
        }

        // ── Recherche avec debounce ───────────────────────────────
        // Le debounce évite de relancer le filtre à chaque frappe :
        // on attend 200ms après la dernière frappe pour filtrer

        let debounceTimer; // Stocke la référence au setTimeout pour pouvoir l'annuler
        if (searchInput) {
            searchInput.addEventListener('input', function () {
                clearTimeout(debounceTimer); // Annule le timer précédent
                debounceTimer = setTimeout(applyFilters, 200); // Lance applyFilters après 200ms de silence
            });

            searchInput.addEventListener('keydown', function (e) {
                if (e.key === 'Escape') { // Si l'utilisateur appuie sur Escape
                    this.value = ''; // Vide le champ de recherche
                    this.blur();     // Enlève le focus du champ
                    applyFilters();  // Réaffiche tous les items
                }
            });
        }

        // ── Filtres par catégorie ──────────────────────────────────

        categoryBtns.forEach(function (btn) {
            btn.addEventListener('click', function () {
                categoryBtns.forEach(function (b) { b.classList.remove('active'); }); // Désactive tous les boutons
                this.classList.add('active'); // Active le bouton cliqué
                applyFilters(); // Relance le filtrage avec la nouvelle catégorie
            });
        });

        // ── Filtrage combiné (recherche + catégorie) ──────────────

        function applyFilters() {
            // Récupère le terme de recherche normalisé en minuscules
            const term = searchInput ? searchInput.value.toLowerCase().trim() : '';

            // Récupère la catégorie active (data-category du bouton actif, ou 'all' si aucun)
            const activeBtn  = document.querySelector('.category-btn.active');
            const category   = activeBtn ? (activeBtn.dataset.category || 'all') : 'all';

            let visibleCount = 0; // Compteur des items visibles après filtrage

            getItems().forEach(function (item) {
                const qEl = item.querySelector('.faq-question-text'); // Élément texte de la question
                const aEl = item.querySelector('.faq-answer');        // Élément texte de la réponse
                if (!qEl || !aEl) return; // Ignore les items mal formés

                const qText  = qEl.textContent.toLowerCase(); // Texte de la question en minuscules
                const aText  = aEl.textContent.toLowerCase(); // Texte de la réponse en minuscules
                const iCat   = item.dataset.category || '';   // Catégorie de cet item (data-category)

                // Vérifie si la catégorie correspond ('all' accepte tout)
                const catOk  = category === 'all' || iCat === category;
                // Vérifie si le terme de recherche est présent dans la question OU la réponse
                const termOk = term === '' || qText.includes(term) || aText.includes(term);
                const show   = catOk && termOk; // L'item est visible seulement si les deux conditions sont vraies

                item.style.display = show ? '' : 'none'; // Affiche ou cache l'item

                // Si l'item est caché mais était ouvert, on le ferme proprement
                if (!show && item.classList.contains('active')) {
                    closeItem(item);
                }

                if (show) visibleCount++; // Incrémente le compteur des items visibles
            });

            // Affiche ou cache le message "aucun résultat" selon le nombre d'items visibles
            noResults.classList.toggle('visible', visibleCount === 0);
        }

        // ── Initialisation ─────────────────────────────────────────

        initItems();    // Configure les accordéons et attache les événements
        applyFilters(); // Applique le filtre initial (affiche tout)
    });

})(); // Fin de l'IIFE

function toggleItem(item) {
  const isOpen  = item.classList.contains('active');
  const content = item.querySelector('.faq-content');
  const trigger = item.querySelector('.faq-question');

  if (isOpen) {
    // ── Fermeture ────────────────────────────────────────────────────
    content.style.maxHeight = '0';
    // maxHeight: 0 → le contenu se replie. Transition CSS prend le relais.

    item.classList.remove('active');
    trigger.setAttribute('aria-expanded', 'false');
  } else {
    // ── Mode exclusif : fermer les autres items ───────────────────────
    if (exclusiveMode) {
      document.querySelectorAll('.faq-item.active').forEach(other => {
        if (other !== item) toggleItem(other);
        // Appel récursif avec l'autre item → le ferme proprement.
      });
    }

    // ── Ouverture ────────────────────────────────────────────────────
    item.classList.add('active');
    content.style.maxHeight = content.scrollHeight + 'px';
    // scrollHeight : hauteur réelle du contenu, même si overflow:hidden.
    // On la lit APRÈS avoir ajouté 'active' (le display n'est plus none).
    // scrollHeight + 'px' = la hauteur exacte nécessaire pour tout afficher.

    trigger.setAttribute('aria-expanded', 'true');
  }
}
