Ce fichier gère la page de finalisation de commande. Il lit le panier et la promo active, calcule les totaux en temps réel selon le mode de livraison choisi, valide le formulaire champ par champ, assemble l'objet commande, l'écrit dans sessionStorage pour confirmation.js, vide le panier, puis redirige.

01 — Numéro de commande

generateOrderNumber() — unique et non-collisionnable ¶

Anatomie d'un numéro de commande généré
CMD-K7X2-1706789012
CMD
préfixe fixe
identifie le type
K7X2
base-36 aléatoire
Math.random().toString(36).slice(2,6).toUpperCase()
1706789012
timestamp Unix
Math.floor(Date.now() / 1000)
La partie aléatoire + le timestamp = collision quasi-impossible même si 2 commandes sont passées au même moment

Chaque commande reçoit un numéro unique combinant un compteur persistant, un timestamp et une part aléatoire, tous encodés en base 36 (0–9, a–z) pour être courts.

javascript
function generateOrderNumber() {
// Compteur persistant : s'incrémente à chaque commande const counter = parseInt(storageGet('peartech-order-counter') || '0', 10) + 1;
storageSet('peartech-order-counter', String(counter));
const timestamp = Date.now().toString(36).toUpperCase(); // ms depuis 1970 en base36 const seq = counter.toString(36).padStart(3, '0').toUpperCase(); // 001, 002… const rand = Math.random().toString(36).substr(2, 3).toUpperCase();
return `CMD-${timestamp}-${seq}${rand}`;
// Ex: CMD-LXYZ9A1B-001K3P
}
Pourquoi base 36 ?
Base 10 (décimal normal)
Date.now() = 1706184930000
13 caractères — long
Base 36 (chiffres + lettres)
1706184930000 "LXYZ9A1B"
8 caractères — compact et lisible
La triple combinaison timestamp + compteur + aléatoire garantit l'unicité même si deux commandes sont passées exactement à la même milliseconde (timestamps identiques compteurs différents, et vice-versa).

storageGet / storageSet / storageRemove — wrappers sécurisés

javascript — accès localStorage protégé par try/catch
function storageGet(key) {
try { return localStorage.getItem(key); }
catch (e) { console.warn('localStorage.getItem échoué :', e); return null; }
}
// localStorage peut échouer si :
// - Navigateur en mode privé avec stockage désactivé
// - Quota dépassé (storageSet)
// - Permissions restreintes dans certains contextes iframe
// Le try/catch évite une erreur fatale qui bloquerait toute la page.

getShippingCost() — lire le prix depuis le DOM ¶

Plutôt que de hardcoder les prix de livraison dans le JS, getShippingCost() les lit directement depuis l'HTML de l'option sélectionnée — si le texte change dans le HTML, le calcul s'adapte automatiquement.

Calcul des montants dans renderSummary() — exemple concret
Sous-total
cart.reduce((acc, item) => acc + item.price * item.quantity, 0) = 1 499 €
Réduction promo
currentPromo ? 1499 × 0.10 = −149.90 € : 0
Livraison
getShippingCost() lu depuis dataset.price du radio sélectionné = 9.90 €
Total
1499 − 149.90 + 9.90 = 1 359 €
renderSummary() est appelée au DOMContentLoaded ET à chaque changement du mode de livraison total toujours à jour sans rechargement.
javascript
function getShippingCost() {
const selected = document.querySelector('input[name="delivery"]:checked');
if (!selected) return 5.90; // Valeur par défaut si rien sélectionné const optionEl = selected.closest('.delivery-option');
const priceSpan = optionEl ? optionEl.querySelector('.option-price') : null;
if (priceSpan) {
const txt = priceSpan.textContent.replace(/[^\d,]/g, '').replace(',', '.');
// "5,90 €"  retire tout sauf chiffres/virgule  "5,90"  replace "," par "."  "5.90" const val = parseFloat(txt);
return isNaN(val) ? 0 : val;
}
// Fallback hardcodé si le span de prix est absent if (selected.value === 'standard') return 5.90;
if (selected.value === 'express')  return 12.90;
return 0;
}
Cascade de fallbacks
input[name="delivery"]:checked l'option radio cochée
si null return 5.90
.closest('.delivery-option') remonte au conteneur parent
si span .option-price trouvé
textContent.replace parseFloat return valeur lue du DOM
si NaN (texte "Gratuit") return 0
fallback hardcodé : 'standard' 5.90, 'express' 12.90

getShippingCost() — Calcul dynamique des frais

javascript
function getShippingCost() {
    const selected = document.querySelector('input[name="delivery"]:checked');
    if (!selected) return 4.99; // Défaut si rien sélectionné

    switch(selected.value) {
        case 'free':     return 0;     // Livraison gratuite (si disponible)
        case 'standard': return 4.99;  // Standard 3-5 jours
        case 'express':  return 9.99;  // Express 24h
        case 'relay':    return 2.99;  // Point relais
        default:         return 4.99;
    }
}
💡
getShippingCost() est appelée à chaque changement de radio input[name='delivery']. C'est une fonction pure conditionnelle — aucun effet de bord, juste une correspondance valeur→prix. Le résultat est immédiatement affiché dans calculateTotals() qui recalcule tout.

function validateForm() {
const required = form.querySelectorAll('input[required], select[required], input[aria-required="true"]');
let valid = true;
required.forEach(field => {
const isEmpty = field.type === 'checkbox' ? !field.checked       // Checkbox : vide = non cochée : !field.value.trim(); // Texte : vide = chaîne vide ou espaces if (isEmpty) {
field.style.borderColor = '#ef4444'; // Rouge valid = false;
} else {
field.style.borderColor = ''; // Rétablit la couleur par défaut CSS }
});
return valid; // true = tout est rempli
}

05 — Soumission finale

submitOrder() — assembler et transmettre la commande ¶

submitOrder() est la fonction terminale de paiement.js : elle assemble toutes les données collectées par le formulaire en un objet orderData structuré, le persiste dans sessionStorage pour confirmation.js, nettoie le panier, puis navigue vers la page de confirmation.

La structure de orderData

javascript
const orderData = {
    orderNumber: generateOrderNumber(),   // ex: 'CMD-1719847234-7K3X'
    email: document.getElementById('email')?.value || '',

    delivery: {
        name:    (prenom + ' ' + nom).trim(),
        address: [adresse, ville, codePostal, pays].filter(Boolean).join(', '),
        mode:    optionEl?.querySelector('.option-title')?.textContent,
        payment: 'Carte bancaire **** 4242' // ou 'PayPal'
    },

    items:         cart.map(i => ({ name, quantity, price })),
    shipping:      getShippingCost(),
    promoCode:     currentPromo?.code    || null,
    promoDiscount: currentPromo ? subtotal * currentPromo.discount : 0,
    total:         subtotal - promoDiscount + shipping
};

generateOrderNumber() — Unicité garantie

La fonction combine Date.now() (timestamp en ms depuis l'epoch Unix) avec une chaîne aléatoire base-36 : 'CMD-' + Date.now() + '-' + Math.random().toString(36).slice(2,6).toUpperCase(). Date.now() garantit l'unicité temporelle (deux commandes ne peuvent pas avoir le même timestamp à la milliseconde). La partie aléatoire ajoute une entropie supplémentaire en cas d'appels simultanés.

[adresse, ville, codePostal, pays].filter(Boolean).join(', ')

Ce pattern construit une adresse complète en ignorant les champs vides. filter(Boolean) supprime les valeurs falsy (null, undefined, chaîne vide) — si codePostal est vide, il n'apparaît pas dans l'adresse finale. join(', ') concatène les valeurs non vides avec une virgule. Résultat propre quels que soient les champs remplis.

?.value — L'opérateur optional chaining

document.getElementById('prenom')?.value utilise l'opérateur optional chaining (ES2020). Si getElementById('prenom') retourne null (l'élément n'existe pas), l'expression s'arrête et retourne undefined au lieu de lever un TypeError: Cannot read properties of null. Le || '' fournit une chaîne vide par défaut.

Le handoff vers confirmation.js

javascript
// Écrire les données pour confirmation.js
try {
    sessionStorage.setItem('peartech-last-order', JSON.stringify(orderData));
} catch(e) {
    console.warn('sessionStorage indisponible :', e);
    // La page de confirmation utilisera ses données de démo
}

// Vider le panier
if (window.PearTechCart) {
    window.PearTechCart.clear();
} else {
    storageRemove('peartech-cart');
    storageSet('peartech-cart-count', '0');
}

// Naviguer
window.location.href = 'page_confirmation.html';

La séquence finale est ordonnée intentionnellement : écrire dans sessionStorage avant de naviguer (pas après), vider le panier avant de naviguer (état cohérent si l'utilisateur appuie sur 'précédent'), puis naviguer. Si l'écriture sessionStorage échoue (try/catch), on navigue quand même — confirmation.js a ses propres données de fallback.

⚠️
Le try/catch autour de sessionStorage.setItem est indispensable. En navigation privée Safari ou sur certains navigateurs en mode kiosque, sessionStorage peut lever une SecurityError ou QuotaExceededError. Sans try/catch, une erreur ici bloquerait la navigation vers la confirmation — l'utilisateur croirait sa commande échouée.

Le handoff sessionStorage vers confirmation.js

Avant de naviguer vers la page de confirmation, paiement.js sérialise toutes les données de la commande dans sessionStorage.setItem('peartech-last-order', JSON.stringify(orderData)). confirmation.js les lit et les consomme immédiatement (read + removeItem). Ce pattern write → navigate → read → delete est la façon standard de passer des données entre deux pages sans backend.

validateForm() — La dernière ligne de défense

Les validations de validateForm() s'exécutent au clic sur 'Confirmer la commande'. Elles vérifient : 1. Tous les champs obligatoires remplis. 2. Format email valide (regex). 3. Format carte bancaire (16 chiffres si paiement CB). 4. Consentement RGPD coché. Si l'une échoue, un message d'erreur ciblé s'affiche et la navigation est bloquée.

✓
Les utilitaires storageGet() et storageSet() wrappent localStorage dans des try/catch. En navigation privée ou stockage plein, localStorage peut lever une exception. Les wrappers garantissent que paiement.js continue de fonctionner même si la persistance est indisponible.

Pourquoi chaque bloc est placé là où il est ¶

paiement.js sépare les utilitaires purs (hors DOM) des comportements liés au DOM (dans DOMContentLoaded). Les actions irréversibles (génération de commande, redirection) sont systématiquement en dernier.

1
Utilitaires storage HORS DOMstorageGet() + storageSet() + storageRemove()
Hors de DOMContentLoaded — utilisées par generateOrderNumber() définie juste après. Ces trois fonctions encapsulent tous les accès localStorage/sessionStorage dans un try/catch. generateOrderNumber() (étape 2) les utilise et est elle aussi hors DOMContentLoaded — les utilitaires doivent donc la précéder.
Utilisées par : generateOrderNumber() (étape 2), chargement du panier (étape 3), et la soumission du formulaire (étape 7).
↓ generateOrderNumber hors DOM — pas besoin du DOM
2
Génération commande HORS DOMfunction generateOrderNumber()
Hors de DOMContentLoaded — ne dépend pas du DOM. Génère un identifiant unique (CMD-TIMESTAMP-SEQ-RAND) en combinant un timestamp en base 36, un compteur séquentiel et des caractères aléatoires. Ne lit ni n'écrit aucun élément HTML — placée hors DOMContentLoaded par nature. Elle est appelée dans la soumission du formulaire (étape 7).
Utilise : storageGet(), storageSet(). Appelée dans : la soumission du formulaire.
↓ dans DOMContentLoaded : lire les données avant d'afficher
3
Chargement panier + promoPearTechCart.getCart() + sessionStorage.getItem('peartech-promo')
En premier dans DOMContentLoaded — données sources de toute la page. Le panier et la promo active sont lus depuis leurs sources respectives. La promo est lue depuis sessionStorage (transmise par cart.js via sessionStorage.setItem('peartech-promo', …) lors du clic 'Commander'). Ces données alimentent l'affichage du résumé et le calcul des totaux.
Alimentent : renderSummary() (étape 5), calculateTotals() (étape 5), et la soumission (étape 7).
↓ récupérer les éléments DOM avant les fonctions d'affichage
4
Récupération DOMgetElementById('summary-items') + … (tous les éléments)
Après les données, avant les fonctions qui les affichent. On sait maintenant ce qu'on va afficher (panier + promo). On récupère les conteneurs qui recevront les données. Cette logique — avoir les données puis les conteneurs, puis afficher — est le pattern de toute la page paiement.
↓ fonctions d'affichage — avant les listeners qui les appellent
5
Affichage résumé + totauxrenderSummary() + calculateTotals()
Définies avant les listeners de livraison/paiement qui les rappellent. renderSummary() liste les articles commandés. calculateTotals() calcule sous-total, frais de livraison, promo et total. Ces deux fonctions sont appelées dès l'initialisation (étape 6) et aussi par les listeners de choix de livraison — elles doivent être définies avant.
calculateTotals() lit : l'option de livraison sélectionnée (querySelector('.delivery-option:checked')), currentPromo. Écrit dans : summaryTotalEl, summaryShipEl, summaryGrandEl.
↓ listeners de livraison/paiement — appellent calculateTotals
6
Options livraison + paiementquerySelectorAll('.delivery-option') + ('.payment-option') + change listeners
Après calculateTotals() qu'ils appellent. Quand l'utilisateur change le mode de livraison (Standard/Express/Click&Collect), calculateTotals() est rappelée pour recalculer le total avec les nouveaux frais. Ces listeners sont statiques — les boutons radio existent dans le HTML — contrairement aux résultats de recherche ou aux cards panier qui sont injectés dynamiquement.
↓ validation et soumission — toujours en dernier
7
Validation + soumissionvalidateForm() + form.addEventListener('submit') — génère commande + redirige
En tout dernier — la soumission est l'action finale et irréversible. Elle valide tous les champs, génère le numéro de commande (generateOrderNumber()), stocke les données dans sessionStorage pour la page de confirmation, vide le panier, et redirige. Ces actions irréversibles (génération, vidage, redirection) doivent être les dernières — elles ne se défont pas.
Les actions irréversibles (générer, vider, rediriger) sont toujours les dernières dans la chaîne d'exécution.

Concepts et termes techniques utilisés dans ce fichier ¶

Définitions des termes importants rencontrés dans cette documentation.

Document Object Model
Représentation en mémoire de la page HTML sous forme d'arbre. Chaque balise devient un nœud. JavaScript modifie cet arbre via getElementById(), querySelector(), innerHTML. Modifier le DOM = modifier ce que l'utilisateur voit en temps réel.
Immediately Invoked Function Expression
Fonction qui s'exécute immédiatement à sa déclaration : (function() { ... })(). Crée un scope isolé — tout ce qui est déclaré dedans reste privé et ne pollue pas window. Tous les fichiers JS du projet utilisent ce pattern.
DOMContentLoaded
Événement déclenché quand le HTML est parsé et le DOM prêt — avant images et CSS. Moment idéal pour initialiser les modules JS. Distinct de window.load qui attend tous les assets réseau.
localStorage
Stockage clé-valeur persistant dans le navigateur — survit à la fermeture. Capacité ~5 Mo. Clé/valeur uniquement en string : les objets passent par JSON.stringify() avant stockage et JSON.parse() à la lecture.
sessionStorage
Identique à localStorage mais limité à l'onglet courant. Effacé à la fermeture de l'onglet. Utilisé ici pour transmettre la commande de paiement.js vers confirmation.js sans la persister.
JSON — JavaScript Object Notation
Format texte pour sérialiser des objets. JSON.stringify(obj) → string stockable. JSON.parse(str) → objet utilisable. Obligatoire pour stocker/lire des objets dans localStorage ou sessionStorage.
dataset
Interface qui expose les attributs data-* d'un élément. el.dataset.productId lit data-product-id. Permet de stocker des données métier directement dans le HTML sans variables globales. La conversion camelCase est automatique : data-product-id → dataset.productId.
URLSearchParams
Lit les paramètres d'une URL : ?id=12&cat=apple. new URLSearchParams(window.location.search).get('id') → '12'. Utilisé sur la page produit pour lire l'ID du produit à afficher depuis l'URL.

Paiement : validation, sécurité et UX critique ¶

💡
Analogie — la validation en couches : La sécurité d'un paiement, c'est comme les portes d'une banque. Première porte : le format (est-ce que c'est un numéro de carte valide ?). Deuxième porte : la logique métier (la date n'est pas expirée ?). Troisième porte : le serveur (la banque accepte-t-elle ?). Passer la première porte ne garantit pas les suivantes.
💡
Analogie — le masquage de carte : Afficher seulement les 4 derniers chiffres d'un numéro de carte, c'est comme montrer seulement la couleur d'une enveloppe sans révéler son contenu. Assez pour identifier ("ma carte bleue se terminant par 4242"), pas assez pour compromettre.
🚨 PIÈGE — Stocker un numéro de carte en localStorage ou variable JS
Ne jamais stocker de données sensibles (numéro carte, CVV) en localStorage, sessionStorage, ou variable globale. Ces données sont accessibles à tout script sur la page, y compris des extensions malveillantes. Dans un vrai projet, ces données ne transitent que vers le serveur de paiement via HTTPS, jamais stockées côté client.
🚨 PIÈGE — Désactiver le bouton "Payer" sans feedback clair
Désactiver le bouton pendant le traitement est correct, mais l'utilisateur doit comprendre pourquoi. Ajouter un spinner et un texte "Traitement en cours…". Sans feedback, l'utilisateur clique plusieurs fois en croyant que ça n'a pas marché — risque de doubles transactions.
PRO TIP
Formate le numéro de carte en temps réel avec des espaces tous les 4 chiffres (value.replace(/\s/g,'').replace(/(\d{4})/g,'$1 ').trim()) et limite à 19 caractères. L'utilisateur reconnaît le format de sa vraie carte.
PRO TIP
Utilise l'algorithme de Luhn pour valider un numéro de carte côté client avant toute soumission. C'est un algorithme simple (somme alternée des chiffres) qui détecte les fautes de frappe sans solliciter le serveur.

Validation format

Regex, longueur, Luhn. Côté client uniquement pour l'UX. Le serveur revalide.

Formatage temps réel

Espaces tous les 4 chiffres, date MM/AA, CVV masqué. Réduit les erreurs de saisie.

États du bouton

Actif → désactivé + spinner → succès ou erreur. Jamais laisser l'état ambigu.

Pas de stockage

Données sensibles : zéro persistence côté client. Transit HTTPS vers serveur uniquement.

// ============================================
// paiement.js - Page de finalisation de commande
// Gère : résumé du panier, options de livraison,
// options de paiement, validation du formulaire
// et redirection vers la confirmation
// ============================================

(function () {
    'use strict';

    // ── Accès localStorage sécurisé (utile pour le compteur de commandes) ──
    function storageGet(key) {
        try {
            return localStorage.getItem(key);
        } catch (e) {
            console.warn('localStorage.getItem échoué :', e);
            return null;
        }
    }

    function storageSet(key, value) {
        try {
            localStorage.setItem(key, value);
            return true;
        } catch (e) {
            console.warn('localStorage.setItem échoué :', e);
            return false;
        }
    }

    function storageRemove(key) {
        try {
            localStorage.removeItem(key);
        } catch (e) {
            console.warn('localStorage.removeItem échoué :', e);
        }
    }

    // ── Génération d'un numéro de commande non-collisionnable ──
    function generateOrderNumber() {
        const counter = parseInt(storageGet('peartech-order-counter') || '0', 10) + 1;
        storageSet('peartech-order-counter', String(counter));

        const timestamp = Date.now().toString(36).toUpperCase();
        const seq       = counter.toString(36).padStart(3, '0').toUpperCase();
        const rand      = Math.random().toString(36).substr(2, 3).toUpperCase();

        return `CMD-${timestamp}-${seq}${rand}`;
    }

    document.addEventListener('DOMContentLoaded', function () {

        // ── Chargement du panier via l'API centralisée ──
        let cart = [];
        if (window.PearTechCart) {
            cart = window.PearTechCart.getCart() || [];
        } else {
            console.warn('PearTechCart non disponible, lecture directe localStorage');
            try {
                cart = JSON.parse(storageGet('peartech-cart')) || [];
            } catch (e) {
                cart = [];
            }
        }

        // ── Récupération de la promo active depuis sessionStorage ──
        let currentPromo = null;
        try {
            const storedPromo = sessionStorage.getItem('peartech-promo');
            if (storedPromo) currentPromo = JSON.parse(storedPromo);
        } catch(e) {
            console.warn('Impossible de lire la promo depuis sessionStorage :', e);
        }

        // ── Récupération des éléments du DOM ──
        const summaryItemsEl  = document.getElementById('summary-items');
        const summaryTotalEl  = document.getElementById('subtotal');
        const summaryShipEl   = document.getElementById('delivery-cost');
        const summaryGrandEl  = document.getElementById('total');
        const form            = document.querySelector('.checkout-form');
        const confirmBtn      = document.getElementById('confirm-order');
        const deliveryOptions = document.querySelectorAll('input[name="delivery"]');
        const paymentOptions  = document.querySelectorAll('input[name="payment"]');
        const paymentDetails  = document.getElementById('cb-fields');

        // ── Affichage du résumé de commande ──
        function renderSummary() {
            if (!summaryItemsEl) return;

            if (cart.length === 0) {
                summaryItemsEl.innerHTML = '<p class="summary-empty">Panier vide.</p>';
                return;
            }

            const subtotal      = cart.reduce((acc, item) => acc + (item.price * item.quantity), 0);
            const promoDiscount = currentPromo ? subtotal * currentPromo.discount : 0;
            const shipping      = getShippingCost();
            const total         = subtotal - promoDiscount + shipping;

            // ── Lignes articles ──
            summaryItemsEl.innerHTML = cart.map(item => {
                const linePrice   = item.price * item.quantity;
                const hasDiscount = item.oldPrice && item.oldPrice > item.price;
                const oldLine     = hasDiscount ? item.oldPrice * item.quantity : null;
                const pct         = hasDiscount ? Math.round((1 - item.price / item.oldPrice) * 100) : 0;
                return `
                <div class="summary-item">
                    <span class="item-name">
                        ${item.quantity}× ${item.name}
                        ${pct > 0 ? `<span class="summary-badge">-${pct}%</span>` : ''}
                    </span>
                    <span class="item-price">
                        ${oldLine ? `<span class="item-old-price">${formatPrice(oldLine)}</span> ` : ''}
                        ${formatPrice(linePrice)}
                    </span>
                </div>`;
            }).join('');

            if (summaryTotalEl) summaryTotalEl.textContent = formatPrice(subtotal);
            if (summaryShipEl)  summaryShipEl.textContent  = shipping === 0 ? 'Gratuit' : formatPrice(shipping);

            const summaryTotals    = document.querySelector('.summary-totals');
            const existingPromoRow = document.getElementById('promo-summary-row');

            if (promoDiscount > 0 && summaryTotals) {
                const label = currentPromo.code
                    ? `🏷️ Code "${currentPromo.code}" (-${Math.round(currentPromo.discount * 100)}%)`
                    : '🏷️ Réduction';

                if (existingPromoRow) {
                    existingPromoRow.querySelector('.promo-label').textContent  = label;
                    existingPromoRow.querySelector('.promo-amount').textContent = `-${formatPrice(promoDiscount)}`;
                } else {
                    const promoRow = document.createElement('div');
                    promoRow.id        = 'promo-summary-row';
                    promoRow.className = 'summary-line';
                    promoRow.style.color = '#10b981';
                    promoRow.innerHTML = `
                        <span class="promo-label">${label}</span>
                        <span class="promo-amount" style="font-weight:700">-${formatPrice(promoDiscount)}</span>
                    `;
                    const totalLine = summaryTotals.querySelector('.summary-line.total');
                    if (totalLine) {
                        summaryTotals.insertBefore(promoRow, totalLine);
                    } else {
                        summaryTotals.appendChild(promoRow);
                    }
                }
            } else if (existingPromoRow) {
                existingPromoRow.remove();
            }

            if (summaryGrandEl) summaryGrandEl.textContent = formatPrice(total);
        }

        function getShippingCost() {
            const selected = document.querySelector('input[name="delivery"]:checked');
            if (!selected) return 5.90;

            const optionEl  = selected.closest('.delivery-option');
            const priceSpan = optionEl ? optionEl.querySelector('.option-price') : null;
            if (priceSpan) {
                const txt = priceSpan.textContent.replace(/[^\d,]/g, '').replace(',', '.');
                const val = parseFloat(txt);
                return isNaN(val) ? 0 : val;
            }
            if (selected.value === 'standard') return 5.90;
            if (selected.value === 'express')  return 12.90;
            return 0;
        }

        function togglePaymentDetails() {
            if (!paymentDetails) return;
            const selected = document.querySelector('input[name="payment"]:checked');
            const isCarte  = selected && selected.value === 'cb';
            paymentDetails.style.display = isCarte ? 'block' : 'none';
        }

        function validateForm() {
            const required = form
                ? form.querySelectorAll('input[required], select[required], input[aria-required="true"]')
                : document.querySelectorAll('input[required], select[required]');
            let valid = true;

            required.forEach(field => {
                const isEmpty = field.type === 'checkbox' ? !field.checked : !field.value.trim();
                if (isEmpty) {
                    field.style.borderColor = '#ef4444';
                    valid = false;
                } else {
                    field.style.borderColor = '';
                }
            });

            return valid;
        }

        function submitOrder() {
            if (!validateForm()) {
                showMessage('Veuillez remplir tous les champs obligatoires.', 'error');
                return;
            }

            if (cart.length === 0) {
                showMessage('Votre panier est vide.', 'error');
                return;
            }

            const selectedDelivery = document.querySelector('input[name="delivery"]:checked');
            const selectedPayment  = document.querySelector('input[name="payment"]:checked');

            const name = (document.getElementById('prenom')?.value || '') + ' '
                       + (document.getElementById('nom')?.value    || '');

            const address = [
                document.getElementById('adresse')?.value,
                document.getElementById('ville')?.value,
                document.getElementById('code-postal')?.value,
                document.getElementById('pays')?.value || 'France'
            ].filter(Boolean).join(', ');

            let paymentLabel = 'Carte bancaire';
            if (selectedPayment?.value === 'paypal') paymentLabel = 'PayPal';
            const cardLast4 = document.getElementById('card-number')?.value.slice(-4);
            if (selectedPayment?.value === 'cb' && cardLast4) {
                paymentLabel = `Carte bancaire **** ${cardLast4}`;
            }

            const subtotal      = cart.reduce((acc, item) => acc + (item.price * item.quantity), 0);
            const promoDiscount = currentPromo ? subtotal * currentPromo.discount : 0;
            const shipping      = getShippingCost();

            const orderData = {
                orderNumber: generateOrderNumber(),
                email: document.getElementById('email')?.value || '',
                delivery: {
                    name:    name.trim() || 'Client',
                    address: address     || '–',
                    mode: (function() {
                        if (!selectedDelivery) return 'Livraison standard (3-5 jours)';
                        const optEl = selectedDelivery.closest('.delivery-option');
                        const t     = optEl ? optEl.querySelector('.option-title') : null;
                        return t ? t.textContent.trim() : 'Livraison standard';
                    })(),
                    payment: paymentLabel
                },
                items:    cart.map(i => ({ name: i.name, quantity: i.quantity, price: i.price })),
                shipping: shipping,
                promoCode:     currentPromo ? currentPromo.code : null,
                promoDiscount: promoDiscount,
                total:    subtotal - promoDiscount + shipping
            };

            try {
                sessionStorage.setItem('peartech-last-order', JSON.stringify(orderData));
            } catch (e) {
                console.warn('sessionStorage indisponible, la confirmation utilisera les données de démo :', e);
            }

            // ── Nettoyage du panier via l'API centralisée ──
            if (window.PearTechCart) {
                window.PearTechCart.clear(); // méthode clear() doit exister
            } else {
                storageRemove('peartech-cart');
                storageSet('peartech-cart-count', '0');
            }

            window.location.href = 'page_confirmation.html';
        }

        function showMessage(text, type) {
            let msg = document.getElementById('paiement-message');
            if (!msg) {
                msg = document.createElement('div');
                msg.id = 'paiement-message';
                msg.style.cssText = 'margin:1rem 0;padding:.9rem 1.2rem;border-radius:8px;font-size:.9rem;font-weight:500;';
                form && form.prepend(msg);
            }
            msg.textContent      = text;
            msg.style.background = type === 'error' ? 'rgba(239,68,68,.1)' : 'rgba(16,185,129,.1)';
            msg.style.border     = `1px solid ${type === 'error' ? '#ef4444' : '#10b981'}`;
            msg.style.color      = type === 'error' ? '#ef4444' : '#10b981';
            msg.scrollIntoView({ behavior: 'smooth', block: 'nearest' });
        }

        function formatPrice(price) {
            return price.toFixed(2).replace('.', ',') + ' €';
        }

        deliveryOptions.forEach(opt => {
            opt.addEventListener('change', function () {
                document.querySelectorAll('.delivery-option').forEach(el => {
                    el.style.borderColor = '';
                    el.style.background  = '';
                });
                const parent = this.closest('.delivery-option');
                if (parent) {
                    parent.style.borderColor = 'var(--primary)';
                    parent.style.background  = 'rgba(59,130,246,.05)';
                }
                renderSummary();
            });
        });

        paymentOptions.forEach(opt => {
            opt.addEventListener('change', togglePaymentDetails);
        });

        if (confirmBtn) {
            confirmBtn.addEventListener('click', function (e) {
                e.preventDefault();
                submitOrder();
            });
        }

        renderSummary();
        togglePaymentDetails();

        if (!document.getElementById('paiement-discount-style')) {
            const style = document.createElement('style');
            style.id = 'paiement-discount-style';
            style.textContent = `
                .item-old-price {
                    text-decoration: line-through;
                    color: var(--text-muted, #9ca3af);
                    font-size: 0.85em;
                    margin-right: 0.25rem;
                }
                .summary-badge {
                    display: inline-block;
                    background: rgba(239, 68, 68, 0.12);
                    color: #ef4444;
                    font-size: 0.72rem;
                    font-weight: 700;
                    padding: 0.1rem 0.4rem;
                    border-radius: 4px;
                    margin-left: 0.3rem;
                    vertical-align: middle;
                }
                .summary-savings .item-name {
                    font-weight: 600;
                }
                .summary-savings {
                    border-top: 1px dashed var(--border, #e5e7eb);
                    padding-top: 0.5rem;
                    margin-top: 0.25rem;
                }
            `;
            document.head.appendChild(style);
        }

        console.log('Paiement initialisé –', cart.length, 'article(s) dans le panier');
    });

})();

① Formatage temps réel du numéro de carte

javascript
cardInput.addEventListener('input', function() {
  let val = this.value
    .replace(/\D/g, '')
    // \D = tout caractère non-digit. Supprime les espaces, tirets, lettres.
    // L'utilisateur peut coller '4242 4242 4242 4242' → on garde '4242424242424242'.

    .slice(0, 16)
    // Maximum 16 chiffres pour une carte Visa/Mastercard.

    .replace(/(\d{4})(?=\d)/g, '$1 ');
    // Regex : groupe de 4 chiffres SUIVI d'un autre chiffre (lookahead ?=\d).
    // $1 = le groupe capturé (les 4 chiffres), ' ' = l'espace ajouté.
    // '1234567890' → '1234 5678 90'
    // Le lookahead évite d'ajouter un espace après le dernier groupe.

  this.value = val;
  // Réassigne la valeur formatée dans l'input.

② Algorithme de Luhn — validation de carte

javascript
function luhn(cardNumber) {
  const digits = cardNumber.replace(/\s/g, '').split('').map(Number);
  // Supprime les espaces, découpe en tableau de chiffres individuels.

  let sum = 0;
  let alt = false; // Alterne : vrai chiffre, double, vrai, double…

  for (let i = digits.length - 1; i >= 0; i--) {
    // On parcourt de droite à gauche.
    let n = digits[i];

    if (alt) {
      n *= 2; // Double un chiffre sur deux
      if (n > 9) n -= 9; // Si résultat > 9 : soustraire 9 (équiv. à sommer les chiffres)
    }

    sum += n;
    alt  = !alt; // Alterne à chaque chiffre
  }

  return sum % 10 === 0;
  // Une carte valide : la somme totale est divisible par 10.
  // 4242424242424242 → true  (numéro de test Stripe ✓)
  // 1234567890123456 → false (numéro invalide)
}

function validatePaymentForm() {
  const card    = cardInput.value.replace(/\s/g, '');
  const expiry  = expiryInput.value; // Format attendu : 'MM/AA'
  const cvv    = cvvInput.value;

  const errors = [];

  // ── Numéro de carte ───────────────────────────────────────────────
  if (!/^\d{16}$/.test(card)) {
    errors.push({ field: 'card', msg: '16 chiffres requis' });
  } else if (!luhn(card)) {
    errors.push({ field: 'card', msg: 'Numéro de carte invalide' });
  }

  // ── Date d'expiration ─────────────────────────────────────────────
  const [mm, yy] = expiry.split('/').map(Number);
  const now      = new Date();
  const expDate  = new Date(2000 + yy, mm - 1);
  // new Date(année, mois-1) : mois 0-indexé. Mois 1 (janvier) = index 0.
  // '12/25' → new Date(2025, 11) = décembre 2025.

  if (!mm || !yy || mm < 1 || mm > 12 || expDate < now) {
    errors.push({ field: 'expiry', msg: 'Date expirée ou invalide' });
  }

  // ── CVV ───────────────────────────────────────────────────────────
  if (!/^\d{3,4}$/.test(cvv)) {
    errors.push({ field: 'cvv', msg: '3 ou 4 chiffres requis' });
  }

  return { valid: errors.length === 0, errors };
}
